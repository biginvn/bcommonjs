BCommonJS

Collect almost useful functions javascript plugin

Github source : https://github.com/biginvn/bcommonjs

Introduction

This plugin base on jQuery library

So we need import jquery before to use it

    
    <script src="https://cdnjs.cloudflare.com/ajax/libs/jquery/2.2.4/jquery.min.js"></script>
    <script src="built/bcommon.js"></script>

Util

currencyFormat( number, name = 'USD', separator = ',' )

Format number to currency markup

                
    var number=12000;
    BLib.currencyFormat(number) // result => 12,000 USD

    BLib.currencyFormat(number, 'VND') // result => 12,000 VND

    BLib.currencyFormat(number, 'SGD', '.') // result => 12.000 SGD
isEmpty( variable )

Check variable empty, it will return true if :

  1. variable = null || undefined
  2. array length = 0
  3. string length = 0
  4. number = 0
                    
    var array=[];
    BLib.isEmpty(array) // result => true;

    var str='';
    BLib.isEmpty(str) // result => true;

    var obj = null;
    BLib.isEmpty(obj) // result => true;

Action

openPopupWindow( link,title, width, height )

Open new window popup auto center screen

                    
    BLib.openPopupWindow( "https://github.com", "Github testing window popup", 300, 300 );
    // This will open new child window base on current window

Animation

scrollToElement( element, delay = 1000 )

Scroll screen to element

                    
    BLib.scrollToElement( '#scroll' );// Scroll viewport to div#scroll
    BLib.scrollToElement( '#scroll', 2000 );// Scroll viewport to div#scroll in 2 seconds

Array

objectInArray (item, array, key)

Return index of item if item.key equals any item of array have the same key => value

                    
    var itemFind = {
        code : 10,
        name : 'item 001'
    };
    var list = [
        {
            code : 10,
            desc : 'the first description',
            name : 'item in list 001'
        },
        {
            code : 13,
            name : 'item in list 002'
        },
        {
            code : 15,
            name : 'item in list 003'
        }
    ];

    BLib.objectInArray( itemFind, list, 'code' ); // result = 0
    BLib.objectInArray( itemFind, list, 'name' ); // result = null - not found
inArrayMultiKey (item, array, keys)

Return index of item if item.key1 , item.key2 ... equals any item of array have the same key => value

                    
    var itemFind = {
        code : 10,
        name : 'item 001'
    };
    var list = [
        {
            code : 10,
            desc : 'the first description',
            name : 'item 001'
        },
        {
            code : 13,
            name : 'item in list 002'
        },
        {
            code : 15,
            name : 'item in list 003'
        }
    ];

    BLib.objectInArray( itemFind, list, ['code', 'name'] ); // result = 0
    BLib.objectInArray( itemFind, list, ['name', 'desc', 'code'] ); // result = null - not found
getObjectItemInArray (list, key, value)

Return object item of array (list) if it's match key <=> value

                    
    var list = [
        {
            code : 10,
            desc : 'the first description',
            name : 'item 001'
        },
        {
            code : 13,
            name : 'item in list 002'
        },
        {
            code : 15,
            name : 'item in list 003'
        }
    ];

    BLib.objectInArray( list, code, 13 ); 
    // result = {
    //                code : 13,
    //                name : 'item in list 002'
    //            }   
    BLib.objectInArray( list, code, 80 ); // result = null - not found
